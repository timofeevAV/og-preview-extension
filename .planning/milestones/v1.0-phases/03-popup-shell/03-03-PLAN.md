**EmptyState.tsx** — Accepts `variant: 'empty' | 'error'` and renders appropriate icon + message. Import hugeicons icons for each variant:
- 'empty': use a document/search icon (e.g. `FileSearchIcon` or `Document01Icon` from `@hugeicons/core-free-icons`) + message "No OG metadata detected"
- 'error': use a lock/shield icon (e.g. `LockIcon` or `Shield01Icon`) + message "Can't access this page"

If a specific icon isn't found in the free pack, use `InformationCircleIcon` as a fallback — it's a standard icon almost certainly in the free tier.

```tsx
import { HugeiconsIcon } from '@hugeicons/react';
import { FileSearchIcon, LockIcon } from '@hugeicons/core-free-icons';
import {
  Empty,
  EmptyContent,
  EmptyDescription,
  EmptyHeader,
  EmptyMedia,
  EmptyTitle,
} from '@/components/ui/empty';

interface EmptyStateProps {
  variant: 'empty' | 'error';
}

export function EmptyState({ variant }: EmptyStateProps) {
  const isError = variant === 'error';
  return (
    <div className="px-4 py-8">
      <Empty>
        <EmptyHeader>
          <EmptyMedia variant="icon">
            <HugeiconsIcon
              icon={isError ? LockIcon : FileSearchIcon}
              size={32}
              color="currentColor"
            />
          </EmptyMedia>
          <EmptyTitle>
            {isError ? "Can't access this page" : 'No OG metadata detected'}
          </EmptyTitle>
          <EmptyDescription>
            {isError
              ? 'The extension cannot read metadata on this page.'
              : 'This page has no Open Graph meta tags.'}
          </EmptyDescription>
        </EmptyHeader>
      </Empty>
    </div>
  );
}
```

**MissingFields.tsx** — Shows list of fields that are absent from ogData. Uses KNOWN_OG_FIELDS filtered to missing keys:
```tsx
import type { OgData } from '@/lib/types';
import { KNOWN_OG_FIELDS } from '@/lib/og-display';

interface MissingFieldsProps {
  ogData: OgData;
}

export function MissingFields({ ogData }: MissingFieldsProps) {
  const missing = KNOWN_OG_FIELDS.filter(f => !ogData[f.key]);

  if (missing.length === 0) return null;

  return (
    <div className="px-3 pb-3">
      <p className="text-xs font-medium text-muted-foreground mb-1.5">Missing fields</p>
      <ul className="space-y-1">
        {missing.map(field => (
          <li key={field.key} className="text-xs text-muted-foreground">
            <span className="font-mono text-foreground/70">{field.label}</span>
            {' — '}
            {field.description}
          </li>
        ))}
      </ul>
    </div>
  );
}
```

**CompactCard.tsx** — Full-width image banner + title + description + expand button:

Read `components/ui/card.tsx` first to understand the exact Card sub-component exports (CardHeader, CardContent, etc.) before writing CompactCard — the mira-style Card API may differ from the standard new-york Card.

```tsx
import { HugeiconsIcon } from '@hugeicons/react';
import { ArrowDown01Icon, ArrowUp01Icon } from '@hugeicons/core-free-icons';
import type { OgData } from '@/lib/types';
import { resolveDisplayData } from '@/lib/og-display';
import { MissingFields } from './MissingFields';

interface CompactCardProps {
  ogData: OgData;
  isPartial: boolean;
  expanded: boolean;
  onToggleExpand: () => void;
}

export function CompactCard({ ogData, isPartial, expanded, onToggleExpand }: CompactCardProps) {
  const { title, description, image } = resolveDisplayData(ogData);

  return (
    <div className="w-[380px]">
      {/* Image banner — full width, 1.91:1 aspect ratio */}
      {image ? (
        <div className="w-full overflow-hidden" style={{ aspectRatio: '1.91' }}>
          <img
            src={image}
            alt={ogData.imageAlt ?? title ?? 'Page preview image'}
            className="w-full h-full object-cover"
          />
        </div>
      ) : (
        <div
          className="w-full bg-muted flex items-center justify-center"
          style={{ aspectRatio: '1.91' }}
        >
          <span className="text-xs text-muted-foreground">No image</span>
        </div>
      )}

      {/* Title + description + expand button */}
      <div className="px-3 pt-2 pb-1">
        {title && (
          <p className="text-sm font-semibold truncate leading-snug">{title}</p>
        )}
        {description && (
          <p className="text-xs text-muted-foreground line-clamp-2 mt-0.5">{description}</p>
        )}
        {ogData.siteName && (
          <p className="text-xs text-muted-foreground/70 mt-0.5 truncate">{ogData.siteName}</p>
        )}
      </div>

      {/* Partial data: show missing fields indicator */}
      {isPartial && <MissingFields ogData={ogData} />}

      {/* Expand / collapse button */}
      <div className="flex justify-center py-1 border-t border-border">
        <button
          onClick={onToggleExpand}
          className="flex items-center gap-1 text-xs text-muted-foreground hover:text-foreground transition-colors px-2 py-0.5 rounded"
          aria-label={expanded ? 'Collapse view' : 'Expand view'}
        >
          <HugeiconsIcon
            icon={expanded ? ArrowUp01Icon : ArrowDown01Icon}
            size={14}
            color="currentColor"
          />
          <span>{expanded ? 'Collapse' : 'Expand'}</span>
        </button>
      </div>
    </div>
  );
}
```

**App.tsx** — State machine orchestrating all states. Read the existing App.tsx before overwriting:

```tsx
import { useEffect, useState } from 'react';
import { sendMessage } from '@/lib/messaging';
import type { OgData } from '@/lib/types';
import { getOgDataStatus } from '@/lib/og-display';
import { OgCardSkeleton } from './components/OgCardSkeleton';
import { EmptyState } from './components/EmptyState';
import { CompactCard } from './components/CompactCard';

export default function App() {
  const [ogData, setOgData] = useState<OgData | null | undefined>(undefined); // undefined = loading
  const [expanded, setExpanded] = useState(false);

  useEffect(() => {
    async function load() {
      try {
        const [tab] = await browser.tabs.query({ active: true, currentWindow: true });
        if (!tab?.id) {
          setOgData(null);
          return;
        }
        const data = await sendMessage('getPageOgData', { tabId: tab.id });
        setOgData(data ?? null);
      } catch {
        setOgData(null);
      }
    }
    load();
  }, []);

  // Loading state (undefined = not yet fetched)
  if (ogData === undefined) {
    return (
      <div className="w-[380px]">
        <OgCardSkeleton />
      </div>
    );
  }

  const status = getOgDataStatus(ogData);

  // Error state (null = sendMessage failed or restricted page)
  if (status === 'error') {
    return (
      <div className="w-[380px]">
        <EmptyState variant="error" />
      </div>
    );
  }

  // Empty state (OgData exists but has no meaningful fields)
  if (status === 'empty') {
    return (
      <div className="w-[380px]">
        <EmptyState variant="empty" />
      </div>
    );
  }

  // Partial or complete — show compact card (ExpandedView added in Plan 04)
  const isPartial = status === 'partial';
  return (
    <div className="w-[380px]">
      <CompactCard
        ogData={ogData}
        isPartial={isPartial}
        expanded={expanded}
        onToggleExpand={() => setExpanded(v => !v)}
      />
      {/* ExpandedView placeholder — wired in Plan 04 */}
      {expanded && (
        <div className="border-t border-border p-4 text-xs text-muted-foreground">
          Expanded view coming soon…
        </div>
      )}
    </div>
  );
}
```

Note on `browser` global: This is the WXT global (no import needed — WXT auto-injects it in entrypoints). The popup `entrypoints/popup/App.tsx` is compiled as a WXT entrypoint so the global is available.

Note on `OgData | null | undefined` for loading state: `undefined` = not yet fetched (show skeleton), `null` = fetch completed with error, `OgData` = fetched successfully (including `{}` for no-tags).
