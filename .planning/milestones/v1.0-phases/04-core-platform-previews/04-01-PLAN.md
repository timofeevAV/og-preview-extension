    Create `entrypoints/popup/components/platform/XCard.tsx`.

    The directory `entrypoints/popup/components/platform/` does not exist yet — create it.

    The X card has two layouts determined by `ogData.twitterCard`:
    - Default / `summary_large_image`: Full-width 16:9 image with gradient overlay; domain + title overlaid at bottom of image
    - `summary`: Horizontal layout — small 80x80px square thumbnail on left, domain + title text on right

    Full implementation:

    ```typescript
    import type { OgData } from '@/lib/types';
    import { resolveDisplayData } from '@/lib/og-display';

    interface PlatformCardProps {
      ogData: OgData;
    }

    function extractDomain(ogData: OgData): string {
      if (ogData.url) {
        try {
          return new URL(ogData.url).hostname.replace(/^www\./, '');
        } catch {
          // malformed URL
        }
      }
      return ogData.siteName ?? '';
    }

    export function XCard({ ogData }: PlatformCardProps) {
      const { title, image } = resolveDisplayData(ogData);
      const domain = extractDomain(ogData);
      const isSummary = ogData.twitterCard === 'summary';

      if (isSummary) {
        // summary: horizontal thumbnail-left layout (~80x80px)
        return (
          <div className="w-full border border-border rounded-2xl overflow-hidden flex flex-row items-stretch">
            <div className="flex-shrink-0 w-[80px] h-[80px] bg-muted">
              {image ? (
                <img
                  src={image}
                  alt={ogData.imageAlt ?? title ?? ''}
                  className="w-full h-full object-cover"
                />
              ) : (
                <div className="w-full h-full bg-muted" />
              )}
            </div>
            <div className="flex flex-col justify-center px-3 py-2 min-w-0">
              {domain && (
                <p className="text-[11px] text-muted-foreground truncate">{domain}</p>
              )}
              {title && (
                <p className="text-[13px] font-medium text-foreground line-clamp-2 leading-snug">{title}</p>
              )}
            </div>
          </div>
        );
      }

      // summary_large_image (default): full-width 16:9 with bottom gradient overlay
      return (
        <div className="w-full rounded-2xl border border-border overflow-hidden">
          <div className="relative aspect-[16/9] bg-muted">
            {image ? (
              <img
                src={image}
                alt={ogData.imageAlt ?? title ?? ''}
                className="w-full h-full object-cover"
              />
            ) : (
              <div className="w-full h-full bg-muted flex items-center justify-center">
                <span className="text-xs text-muted-foreground">No image</span>
              </div>
            )}
            {/* Gradient overlay — always rendered so domain/title text is legible on any image */}
            <div className="absolute inset-0 bg-gradient-to-t from-black/60 via-black/0 to-transparent" />
            {/* Domain + title over image at bottom */}
            <div className="absolute bottom-0 left-0 right-0 px-3 py-2">
              {domain && (
                <p className="text-[11px] text-white/60 truncate">{domain}</p>
              )}
              {title && (
                <p className="text-[13px] font-medium text-white truncate">{title}</p>
              )}
            </div>
          </div>
        </div>
      );
    }
    ```

    Do NOT use hardcoded hex colors — use Tailwind tokens (`text-muted-foreground`, `bg-muted`, `border-border`) for light/dark compatibility. The white overlay text is intentionally white on both themes because it sits on a dark gradient over the image.

    Do NOT add description to XCard — X only shows domain + title in the feed for summary_large_image (description is omitted per research).
