import type { OgData } from '@/lib/types';
import { resolveDisplayData } from '@/lib/og-display';

interface TooltipCardProps {
  ogData: OgData;
  url: string;
}

function extractDomain(url: string): string {
  try { return new URL(url).hostname.replace(/^www\./, ''); } catch { return url; }
}

export function TooltipCard({ ogData, url }: TooltipCardProps) {
  const { title, description, image } = resolveDisplayData(ogData);
  const domain = extractDomain(url);

  return (
    <div className="w-full">
      {image && (
        <div className="w-full h-[90px] overflow-hidden">
          <img
            src={image}
            alt={ogData.imageAlt ?? title ?? ''}
            className="w-full h-full object-cover"
          />

**Create `components/tooltip/OgTooltip.tsx`:**

Positioning shell with viewport clamping and opacity fade-in. Uses fixed width of 320px. Two-pass positioning: render invisible first, measure height, then clamp and make visible:

```tsx
import { useRef, useLayoutEffect, useState } from 'react';
import type { OgData } from '@/lib/types';
import { TooltipSkeleton } from './TooltipSkeleton';
import { TooltipErrorState } from './TooltipErrorState';
import { TooltipCard } from './TooltipCard';

const TOOLTIP_WIDTH = 320;
const MARGIN = 8;

interface OgTooltipProps {
  x: number;
  y: number;
  phase: 'loading' | 'ready' | 'error';
  data?: OgData;
  url: string;
}

export function OgTooltip({ x, y, phase, data, url }: OgTooltipProps) {
  const [pos, setPos] = useState<{ left: number; top: number; visible: boolean }>({
    left: x,
    top: y + MARGIN,
    visible: false,
  });
  const ref = useRef<HTMLDivElement>(null);

  useLayoutEffect(() => {
    if (!ref.current) return;
    const { height } = ref.current.getBoundingClientRect();
    const vw = window.innerWidth;
    const vh = window.innerHeight;

    // Clamp horizontal so tooltip stays inside viewport
    const left = Math.max(MARGIN, Math.min(x, vw - TOOLTIP_WIDTH - MARGIN));

    // Default: below the link
    let top = y + MARGIN;

    // Flip above if not enough space below
    if (top + height > vh - MARGIN) {
      top = y - height - MARGIN;
    }

    // Final vertical clamp (last resort)
    top = Math.max(MARGIN, Math.min(top, vh - height - MARGIN));

    setPos({ left, top, visible: true });
  }, [x, y, phase]); // Re-clamp when phase changes (skeleton → card have different heights)

  return (
    <div
      ref={ref}
      className="og-tooltip-card fixed z-[2147483647] w-[320px] rounded-lg border bg-card text-card-foreground shadow-md overflow-hidden"
      style={{
        left: pos.left,
        top: pos.top,
        opacity: pos.visible ? 1 : 0,
        transition: 'opacity 0.1s ease',
      }}
    >
      {phase === 'loading' && <TooltipSkeleton />}
      {phase === 'error' && <TooltipErrorState />}
      {phase === 'ready' && data && <TooltipCard ogData={data} url={url} />}
    </div>
  );
}
```

Important: `og-tooltip-card` class enables `pointer-events: auto` (defined in tooltip/style.css). Without this, mouse events over the tooltip card itself would pass through to the page. The `fixed` class on the inner div requires `@property` for border and shadow — WXT 0.20.17 auto-extracts `@property` to document head so this works correctly.

Replace the stub `entrypoints/tooltip/TooltipApp.tsx` with the full state machine:

- Manages state: `hidden | loading | ready | error`
- `show(url, x, y)`: cancels any in-flight stale-fetch guard, sets loading state, calls `sendMessage('getOgData', { url })`, transitions to `ready` or `error` based on result
- `hide()`: resets state to hidden, clears stale-fetch guard
- Wires `controllerRef.show` and `controllerRef.hide` on every render (ref mutation pattern — safe, avoids stale closure)
- Returns `null` when hidden, renders `<OgTooltip>` when loading/ready/error

```tsx
import { useState, useCallback, useRef } from 'react';
import { sendMessage } from '@/lib/messaging';
import { OgTooltip } from '@/components/tooltip/OgTooltip';
import type { OgData } from '@/lib/types';

export interface Controller {
  show: (url: string, x: number, y: number) => void;
  hide: () => void;
}

type TooltipState =
  | { phase: 'hidden' }
  | { phase: 'loading'; url: string; x: number; y: number }
  | { phase: 'ready'; url: string; x: number; y: number; data: OgData }
  | { phase: 'error'; url: string; x: number; y: number };

interface TooltipAppProps {
  controllerRef: Controller;
}

export function TooltipApp({ controllerRef }: TooltipAppProps) {
  const [state, setState] = useState<TooltipState>({ phase: 'hidden' });
  // staleRef tracks the current fetch so we can discard results from superseded fetches
  const staleRef = useRef(0);

  const show = useCallback(async (url: string, x: number, y: number) => {
    const fetchId = ++staleRef.current;
    setState({ phase: 'loading', url, x, y });

    try {
      const data = await sendMessage('getOgData', { url });
      // Discard if a newer show() call arrived while we were fetching
      if (staleRef.current !== fetchId) return;
      if (data && (data.title || data.description || data.image)) {
        setState({ phase: 'ready', url, x, y, data });
      } else {
        setState({ phase: 'error', url, x, y });
      }
    } catch {
      if (staleRef.current === fetchId) {
        setState({ phase: 'error', url, x, y });
      }
    }
  }, []);

  const hide = useCallback(() => {
    staleRef.current++; // invalidate any in-flight fetch
    setState({ phase: 'hidden' });
  }, []);

  // Mutate controller ref in-place so event listeners always call current functions
  controllerRef.show = show;
  controllerRef.hide = hide;

  if (state.phase === 'hidden') return null;

  return (
    <OgTooltip
      x={state.x}
      y={state.y}
      phase={state.phase}
      data={'data' in state ? state.data : undefined}
      url={state.url}
    />
  );
}
```

Note on stale fetch handling: `@webext-core/messaging` v2.3.0 does not support AbortSignal, so the background fetch runs to completion even after `hide()`. The `staleRef` counter prevents stale results from updating state after the tooltip is hidden or a new URL was requested.

Note on empty OgData: the background returns `{}` (empty OgData object) when a page has no OG tags. Check `data.title || data.description || data.image` to distinguish meaningful data from an empty object before setting `ready` state; fall through to `error` state for empty objects.
